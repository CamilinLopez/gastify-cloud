Table estado_cilindros {
  id int [primary key]
  tipo varchar
}

Table tipo_cilindros {
  id int [primary key]
  tipo varchar
}

Table camiones {
  id int [primary key]
  marca varchar 
  modelo varchar
  placa varchar
  capacidad_carga int
}

Table conductores {
  id int [primary key]
  nombre varchar
  licencia varchar
  fecha_registro date
}

Table inventario_camiones {
  id int [primary key]
  camionId int
  tipoCilindroId int
  estadoCilindroId int
  cantidad int
}

Table conductor_camion {
  id int [primary key]
  camionId int
  conductorId int 
  fecha_asignacion date
  hora_asignacion time
}

Ref: camiones.id < inventario_camiones.camionId
Ref: tipo_cilindros.id < inventario_camiones.tipoCilindroId
Ref: estado_cilindros.id < inventario_camiones.estadoCilindroId
Ref: conductores.id < conductor_camion.conductorId
Ref: camiones.id < conductor_camion.camionId
